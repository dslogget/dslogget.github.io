198 void transistorTestControl( T V_gs_in, T V_ds_in, TransistorTestResult< T > & toRet ) {
199  constexpr T alpha = 1.3;
200  constexpr T beta0 = 0.42;
201  constexpr T gamma = 0.0005;
202  constexpr T delta = 0.3;
203  constexpr T xi = 0.06;
204  constexpr T lambda = 1.5;
205  constexpr T mu = 0.0;
206  constexpr T zeta = 0.18;
207  constexpr T Vto = -2.4;
208 
209  T V_gs = V_gs_in;
210  T V_ds = V_ds_in;
211 
212  auto beta = beta0;
213  auto Vgst = V_gs - ( 1 + beta*beta ) * Vto + gamma * V_ds;
214  auto Veff = 0.5 * ( Vgst + std::pow( std::pow( Vgst, 2 ) + delta*delta, 0.5 ) );
215  auto power = lambda / ( 1 + mu * std::pow( V_ds, 2 ) + xi * Veff );
216  auto area = alpha * V_ds * ( 1 + zeta * Veff );
217  auto f1 = std::tanh( area );
218  auto Ids_lim = beta * std::pow( Veff, power );
219  auto Idrain = Ids_lim * f1;
220  toRet.var = Idrain;
221 
222  auto dVeff_dVgs = 0.5*( 1 + Vgst * std::pow( Vgst * Vgst + delta * delta, -0.5 ) );
223  auto dpower_dVgs = -lambda * xi * dVeff_dVgs * std::pow( power / lambda, 2 );
224  auto df_dVgs = std::pow(1 / cosh(area), 2) * alpha * V_ds * zeta * dVeff_dVgs;
225  toRet.diff1 = Idrain * ( power * ( dVeff_dVgs / Veff ) + std::log( Veff ) * dpower_dVgs ) + Ids_lim * df_dVgs ;
226 
227  auto dVeff_dVds = 0.5 * ( gamma + std::pow( Vgst * Vgst + delta * delta, -0.5 ) * Vgst *gamma );
228  auto dpower_dVds = -lambda * ( 2 * mu * V_ds + xi * dVeff_dVds ) * std::pow( power / lambda, 2 );
229  auto df_dVds = std::pow(1 / cosh(area), 2) * alpha * ( 1 + zeta * ( V_ds * dVeff_dVds + Veff ) );
230  toRet.diff2 = Idrain *( power * ( dVeff_dVds / Veff ) + std::log( Veff ) * dpower_dVds ) + Ids_lim * df_dVds;
231 }
